Conditional Forms

  1. Home
  2. Examples
  3. Conditional Forms

You can use JSON Logic to create complex conditionals and validation criteria.

Formio.createForm(document.getElementById('formio'), {
  components: [
    {
      "type": "textfield",
      "label": "Type \"Show\"",
      "description": "Must be exact, and case sensitive.",
      "key": "typeShow",
      "input": true,
      "inputType": "text",
      "validate": {
        "json": {
          "if": [
            {
              "==": [
                {
                  "var": "data.typeShow"
                },
                "Show"
              ]
            },
            true,
            "You must type 'Show'!"
          ]
        }
      }
    },
    {
